options=(   возможность выбора опций
	'1'
	"2"
	'3'
	'4'
	'5'
)
echo 'Please choose an option'  Заголовок
select input in "${options[@]}"; do сюда вставляем значения из оптионс, в зависимости от значения меняется эффект
	case "$input" in
		"${options[0]}")  выводит список всех файлов
			ls
			;;
		"${options[1]}")  выводит дату и время
			date $'+Time: %T\nDate: %D форматируем дату и время
			;;
		"${options[2]}") 
			read -p 'What file do you wish to check? ' finput при выборе этой опции получим сообщение "Какой файл хотите чекнуть?", Shellcheck говорит что без -r будет искажать обратные косые черты
			if [[ -d $finput ]]; then
				format='%s is a directory.\n' проверка, является ли файл директорией.
			elif [[ -f $finput ]]; then
				format='%s is a file.\n' проверка, является ли файл файлом
			else
				format='%s does not exist.\n' проверка, существует ли файл 
			fi
			printf "$format" "$finput"
			;;
		"${options[3]}")
			read -p 'Please enter filename: ' binput Выводит сообщение "Введите имя файла" Shellcheck говорит что без -r будет искажать обратные косые черты
			cp "$binput"{,.bak}
			;;
		"${options[4]}")
			break Прерывает цикл
			;;
		*)
			echo 'Not a valid option.'
			;;
	esac
done скрипт как минимум с ошибками
